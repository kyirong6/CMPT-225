S(n) {
	if (n=0) {
		r = 0
	else {
		r = n + S(n-1)
	}
	return r;
}

Sum(L) {
	if L.empty
		return 0;
	else 
		return L.first + Sum(L.rest);
}

Max(L){
	if L.rest().empty()
		return L.first
	else
		return max(L.first, Max(L.rest()))
}
